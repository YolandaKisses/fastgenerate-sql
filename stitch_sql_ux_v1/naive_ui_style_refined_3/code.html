-- FastGenerate SQL 生成: 统计 2023 年度各季度产品收入 SELECT DATE_TRUNC('quarter', created_at) AS revenue_quarter, category_name, SUM(total_amount) AS total_revenue, COUNT(order_id) AS transaction_count FROM ecommerce.orders JOIN ecommerce.products USING (product_id) WHERE created_at BETWEEN '2023-01-01' AND '2023-12-31' GROUP BY 1, 2 ORDER BY 1 DESC, 3 DESC;
